# 路由设计

## 🎯 概述

本项目使用 **Vue Router 4** 进行路由管理，实现单页应用（SPA）的页面导航和权限控制。

---

## 📂 路由结构

```
/                              # 根路径，重定向到行情看板
├── /login                     # 登录页
├── /                           # 主布局（需要登录）
│   ├── /                       # 行情看板（默认页）
│   ├── /market                 # 行情相关
│   │   ├── /dashboard          # 行情看板
│   │   └── /detail/:symbol     # 行情详情
│   ├── /strategy               # 策略相关
│   │   ├── /list               # 策略列表
│   │   ├── /create             # 创建策略
│   │   └── /edit/:id           # 编辑策略
│   ├── /runtime                # 策略运行
│   │   └── /:instanceId        # 运行详情
│   ├── /monitor                # 监控中心
│   │   └── /dashboard          # 监控面板
│   └── /settings               # 设置
│       ├── /account            # 账户配置
│       └── /system             # 系统设置
└── /404                        # 404 页面
```

## 🛡️ 路由守卫

### 守卫配置 (guards.ts)

```typescript
// src/router/guards.ts
import type { Router } from 'vue-router'
import { useUserStore } from '@/stores/user'
import { ElMessage } from 'element-plus'

/**
 * 路由前置守卫
 */
export function setupRouterGuard(router: Router) {
  router.beforeEach(async (to, from, next) => {
    // 设置页面标题
    document.title = `${to.meta.title || '页面'} - 量化交易系统`

    const userStore = useUserStore()

    // 检查是否需要登录
    if (to.meta.requiresAuth !== false && !userStore.isLoggedIn) {
      // 未登录，跳转到登录页
      ElMessage.warning('请先登录')
      next({
        path: '/login',
        query: { redirect: to.fullPath } // 保存跳转目标
      })
      return
    }

    // 如果已登录且访问登录页，跳转到首页
    if (to.path === '/login' && userStore.isLoggedIn) {
      next({ path: '/' })
      return
    }

    // 检查权限（如果需要）
    if (to.meta.permission && !userStore.hasPermission(to.meta.permission as string)) {
      ElMessage.error('没有权限访问该页面')
      next(false)
      return
    }

    next()
  })

  router.afterEach((to) => {
    // 可以在这里做一些页面加载后的操作
    // 例如：关闭 loading、埋点统计等
  })

  router.onError((error) => {
    console.error('Router error:', error)
    ElMessage.error('页面加载失败')
  })
}
```

## 🔗 路由入口 (index.ts)

```typescript
// src/router/index.ts
import { createRouter, createWebHistory } from 'vue-router'
import routes from './routes'
import { setupRouterGuard } from './guards'

const router = createRouter({
  history: createWebHistory(import.meta.env.BASE_URL),
  routes
})

// 设置路由守卫
setupRouterGuard(router)

export default router
```

## 📋 路由元信息 (meta)

### 元信息字段

| 字段 | 类型 | 说明 | 默认值 |
|------|------|------|--------|
| `title` | string | 页面标题 | - |
| `icon` | string | 菜单图标 | - |
| `requiresAuth` | boolean | 是否需要登录 | true |
| `permission` | string | 所需权限 | - |
| `layout` | string | 布局类型（default/blank） | default |
| `hidden` | boolean | 是否隐藏（不显示在菜单） | false |
| `affix` | boolean | 是否固定在标签栏 | false |
| `cache` | boolean | 是否缓存 | false |

### 使用示例

```typescript
{
  path: '/strategy/list',
  name: 'StrategyList',
  component: () => import('@/views/strategy/StrategyList.vue'),
  meta: {
    title: '策略列表',
    icon: 'List',
    requiresAuth: true,
    permission: 'strategy:read',
    cache: true, // 缓存页面
    affix: true  // 固定在标签栏
  }
}
```

```vue
<script setup lang="ts">
```

## 🔗 路由跳转方式

### 1. 声明式导航

```vue
<template>
  <!-- 基本跳转 -->
  <router-link to="/strategy/list">策略列表</router-link>

  <!-- 带参数 -->
  <router-link :to="{ name: 'StrategyEdit', params: { id: '123' }}">
    编辑策略
  </router-link>

  <!-- 带查询参数 -->
  <router-link :to="{ path: '/strategy/list', query: { status: 'active' }}">
    活跃策略
  </router-link>

  <!-- 动态路由 -->
  <router-link :to="`/market/detail/${symbol}`">
    查看详情
  </router-link>
</template>
```

### 2. 编程式导航

```typescript
import { useRouter, useRoute } from 'vue-router'

const router = useRouter()
const route = useRoute()

// 跳转到指定路由
router.push('/strategy/list')

// 带参数跳转
router.push({
  name: 'StrategyEdit',
  params: { id: '123' }
})

// 带查询参数跳转
router.push({
  path: '/strategy/list',
  query: { page: 1, pageSize: 10 }
})

// 替换当前路由（不产生历史记录）
router.replace('/market/dashboard')

// 返回上一页
router.back()

// 前进一页
router.forward()

// 前进/后退 n 页
router.go(-1)

// 获取当前路由信息
console.log(route.path)      // 当前路径
console.log(route.params)     // 路由参数
console.log(route.query)      // 查询参数
console.log(route.meta)       // 路由元信息
```

### 3. 返回登录页并保存跳转目标

```typescript
const handleLogout = () => {
  const router = useRouter()
  const userStore = useUserStore()

  userStore.logout()
  router.push({ path: '/login', query: { redirect: router.currentRoute.value.fullPath } })
}
```

## 🔐 权限路由

### 路由权限控制

```typescript
{
  path: '/admin',
  name: 'Admin',
  component: AdminPage,
  meta: {
    requiresAuth: true,
    permission: 'admin:access' // 需要管理员权限
  }
}

// 守卫中检查
router.beforeEach((to, from, next) => {
  const userStore = useUserStore()

  if (to.meta.permission && !userStore.hasPermission(to.meta.permission as string)) {
    ElMessage.error('没有权限访问该页面')
    next(false)
    return
  }

  next()
})
```

### 动态路由（可选）

如果需要根据用户权限动态加载路由：

```typescript
// 获取用户权限后动态添加路由
const addDynamicRoutes = (permissions: string[]) => {
  const dynamicRoutes: RouteRecordRaw[] = []

  if (permissions.includes('admin:access')) {
    dynamicRoutes.push({
      path: '/admin',
      name: 'Admin',
      component: () => import('@/views/admin/AdminPage.vue'),
      meta: {
        title: '管理后台',
        permission: 'admin:access'
      }
    })
  }

  dynamicRoutes.forEach(route => router.addRoute(route))
}
```

## 📝 最佳实践

### 1. 路由懒加载

所有页面组件都使用动态导入，实现代码分割：

```typescript
{
  path: '/strategy/list',
  component: () => import('@/views/strategy/StrategyList.vue')
}
```

### 2. 路由命名

使用有意义的路由名称，便于编程式导航：

```typescript
{
  path: '/strategy/list',
  name: 'StrategyList',  // ✅ 使用命名路由
  // name: 'list'         // ❌ 名称不清晰
  component: StrategyList
}
```

### 3. 路由分组

将相关路由分组，便于维护：

```typescript
{
  path: '/strategy',
  name: 'Strategy',
  redirect: '/strategy/list',
  children: [
    { path: 'list', name: 'StrategyList', component: StrategyList },
    { path: 'create', name: 'StrategyCreate', component: StrategyCreate },
    { path: 'edit/:id', name: 'StrategyEdit', component: StrategyEdit }
  ]
}
```

### 4. 路由元信息

使用 meta 存储页面相关信息：

```typescript
{
  path: '/strategy/list',
  meta: {
    title: '策略列表',
    icon: 'List',
    cache: true,
    affix: true
  }
}
```

### 5. 路由守卫分离

将路由守卫逻辑分离到单独文件：

```typescript
// src/router/guards.ts
export function setupRouterGuard(router: Router) {
  router.beforeEach(...)
  router.afterEach(...)
}

// main.ts
import { setupRouterGuard } from '@/router/guards'
setupRouterGuard(router)
```
