**示例：**

```typescript
// src/api/market.ts
import request from './request'
import type { Ticker } from './types'

export const getTickers = (symbols: string[]) => {
  return request.get<Ticker[]>('/api/market/tickers', {
    params: { symbols }

  return request.post('/api/market/subscribe', { symbols })
```

### 3. `/src/composables` - 组合式函数

可复用的逻辑封装，利用 Vue 3 Composition API。

**职责：**
- 状态逻辑复用
- 副作用封装（WebSocket, 轮询等）
- 业务逻辑抽象

**示例：**

```typescript
// src/composables/useWebSocket.ts
import { ref, onUnmounted } from 'vue'

export function useWebSocket(url: string) {
  const ws = ref<WebSocket | null>(null)
  const connected = ref(false)
  const data = ref<any>(null)

  const connect = () => {
    ws.value = new WebSocket(url)

    ws.value.onopen = () => {
      connected.value = true
    }

    ws.value.onmessage = (event) => {
      data.value = JSON.parse(event.data)
    }

    ws.value.onerror = (error) => {
      console.error('WebSocket error:', error)
    }
  }

  const send = (message: any) => {
    if (ws.value?.readyState === WebSocket.OPEN) {
      ws.value.send(JSON.stringify(message))
    }
  }

  const disconnect = () => {
    ws.value?.close()
    connected.value = false
  }

  onUnmounted(() => {
    disconnect()
  })

  return {
    ws,
    connected,
    data,
    connect,
    send,
    disconnect
  }
}
```

**示例：**

```typescript
// src/stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  const token = ref<string>('')
  const userInfo = ref<UserInfo | null>(null)

  const isLoggedIn = computed(() => !!token.value)

  const login = async (credentials: LoginCredentials) => {
    const res = await authApi.login(credentials)
    token.value = res.token
    userInfo.value = res.user
  }

  const logout = () => {
    token.value = ''
    userInfo.value = null
  }

  return {
    token,
    userInfo,
    isLoggedIn,
    login,
    logout
  }
})
```

```typescript
// src/utils/format.ts
export function formatPrice(price: number, decimals = 2): string {
  return price.toFixed(decimals)
}

export function formatPercentage(value: number): string {
  const sign = value >= 0 ? '+' : ''
  return `${sign}${value.toFixed(2)}%`
}

export function formatVolume(volume: number): string {
  if (volume >= 1e9) return `${(volume / 1e9).toFixed(2)}B`
  if (volume >= 1e6) return `${(volume / 1e6).toFixed(2)}M`
  if (volume >= 1e3) return `${(volume / 1e3).toFixed(2)}K`
  return volume.toString()
}
```

## 🎯 命名规范

### 文件命名

- 组件文件：PascalCase，如 `UserProfile.vue`
- 工具文件：camelCase，如 `formatDate.ts`
- 常量文件：UPPER_SNAKE_CASE，如 `API_BASE_URL.ts`
- 类型文件：camelCase，如 `userTypes.ts`

### 组件命名

```vue
<!-- ✅ 正确 -->
<template>
  <UserProfile />

<template>
```

## 📚 最佳实践

### 1. 组件拆分原则

- 单一职责：每个组件只做一件事
- 适度拆分：避免过度拆分或过度耦合
- 可复用优先：提取可复用的逻辑和UI

### 2. 代码组织

- 相关文件放在一起（组件、样式、测试）
- 按功能模块组织，而非按技术层级
- 导入顺序：Vue → 第三方库 → 本地模块

### 3. 性能优化

- 使用 `v-memo` 减少不必要的渲染
- 大列表使用虚拟滚动
- 图片懒加载
- 路由懒加载
