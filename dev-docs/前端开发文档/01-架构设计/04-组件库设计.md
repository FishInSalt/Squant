## 📦 Element Plus 配置

### 安装与引入

```typescript
// src/main.ts
import { createApp } from 'vue'
import ElementPlus from 'element-plus'
import 'element-plus/dist/index.css'
import * as ElementPlusIconsVue from '@element-plus/icons-vue'

const app = createApp(App)

// 引入 Element Plus
app.use(ElementPlus)

// 注册所有图标
for (const [key, component] of Object.entries(ElementPlusIconsVue)) {
  app.component(key, component)
}

app.mount('#app')
```

```vue
<!-- components/common/DataTable.vue -->
<script setup lang="ts">
import { computed, ref, watch } from 'vue'
import type { TableColumnCtx } from 'element-plus'
import { ElTable, ElTableColumn, ElPagination } from 'element-plus'

interface Props {
  data: any[]
  columns: any[]
  loading?: boolean
  pagination?: {
    page: number
    pageSize: number
    total: number
  }
  selectable?: boolean
  stripe?: boolean
  border?: boolean
}

interface Emits {
  (e: 'selection-change', selection: any[]): void
  (e: 'page-change', page: number, pageSize: number): void
}

const props = withDefaults(defineProps<Props>(), {
  loading: false,
  selectable: false,
  stripe: true,
  border: false
})

const emit = defineEmits<Emits>()

const tableRef = ref()

const handleSelectionChange = (selection: any[]) => {
  emit('selection-change', selection)
}

const handlePageChange = (page: number, pageSize: number) => {
  emit('page-change', page, pageSize)
}

// 清空选择
const clearSelection = () => {
  tableRef.value?.clearSelection()
}

// 导出方法给父组件
defineExpose({
  clearSelection
})
</script>

<template>
  <div class="data-table">
    <ElTable
      ref="tableRef"
      :data="data"
      :loading="loading"
      :stripe="stripe"
      :border="border"
      @selection-change="handleSelectionChange"
    >
      <ElTableColumn
        v-if="selectable"
        type="selection"
        width="55"
      />
      <ElTableColumn
        v-for="col in columns"
        :key="col.prop"
        :prop="col.prop"
        :label="col.label"
        :width="col.width"
        :min-width="col.minWidth"
        :fixed="col.fixed"
        :sortable="col.sortable"
        :align="col.align || 'center'"
      >
        <template #default="{ row, column, $index }" v-if="col.slot">
          <slot :name="col.slot" :row="row" :column="column" :index="$index" />
        </template>
      </ElTableColumn>
    </ElTable>

    <div v-if="pagination" class="pagination-wrapper">
      <ElPagination
        :current-page="pagination.page"
        :page-size="pagination.pageSize"
        :total="pagination.total"
        :page-sizes="[10, 20, 50, 100]"
        layout="total, sizes, prev, pager, next, jumper"
        @current-change="handlePageChange"
        @size-change="handlePageChange"
      />
    </div>
  </div>
</template>

<style scoped lang="scss">
.data-table {
  .pagination-wrapper {
    margin-top: 20px;
    display: flex;
    justify-content: flex-end;
  }
}
</style>
```

### 2. 搜索表单组件

```vue
<!-- components/common/SearchForm.vue -->
<script setup lang="ts">
import { ref, watch, reactive } from 'vue'
import { ElForm, ElFormItem, ElInput, ElSelect, ElDatePicker, ElButton } from 'element-plus'

interface FormItem {
  prop: string
  label: string
  type: 'input' | 'select' | 'date'
  options?: { label: string; value: any }[]
  placeholder?: string
  props?: any
}

interface Props {
  fields: FormItem[]
}

interface Emits {
  (e: 'search', params: Record<string, any>): void
  (e: 'reset'): void
}

const props = defineProps<Props>()
const emit = defineEmits<Emits>()

const form = reactive<Record<string, any>>({})

const handleSearch = () => {
  emit('search', form)
}

const handleReset = () => {
  Object.keys(form).forEach(key => {
    form[key] = ''
  })
  emit('reset')
}
</script>

<template>
  <div class="search-form">
    <ElForm :model="form" inline>
      <ElFormItem
        v-for="field in fields"
        :key="field.prop"
        :label="field.label"
        :prop="field.prop"
      >
        <ElInput
          v-if="field.type === 'input'"
          v-model="form[field.prop]"
          :placeholder="field.placeholder"
          v-bind="field.props"
        />
        <ElSelect
          v-if="field.type === 'select'"
          v-model="form[field.prop]"
          :placeholder="field.placeholder"
          v-bind="field.props"
        >
          <ElOption
            v-for="opt in field.options"
            :key="opt.value"
            :label="opt.label"
            :value="opt.value"
          />
        </ElSelect>
        <ElDatePicker
          v-if="field.type === 'date'"
          v-model="form[field.prop]"
          type="daterange"
          v-bind="field.props"
        />
      </ElFormItem>
      <ElFormItem>
        <ElButton type="primary" @click="handleSearch">搜索</ElButton>
        <ElButton @click="handleReset">重置</ElButton>
      </ElFormItem>
    </ElForm>
  </div>
</template>

<style scoped lang="scss">
.search-form {
  padding: 20px;
  background: #f5f7fa;
  border-radius: 4px;
  margin-bottom: 20px;
}
</style>
```

### 4. K线图组件

```vue
<!-- components/charts/KLineChart.vue -->
<script setup lang="ts">
import { ref, onMounted, onUnmounted, watch } from 'vue'
import { createChart, ColorType, CrosshairMode } from 'lightweight-charts'
import type { KLine } from '@/types/market'

interface Props {
  data: KLine[]
  height?: number
  symbol?: string
}

const props = withDefaults(defineProps<Props>(), {
  height: 400
})

const chartContainer = ref<HTMLElement>()
let chart: any = null
let candleSeries: any = null

const initChart = () => {
  if (!chartContainer.value) return

  chart = createChart(chartContainer.value, {
    width: chartContainer.value.clientWidth,
    height: props.height,
    layout: {
      background: { type: ColorType.Solid, color: 'transparent' },
      textColor: '#333'
    },
    grid: {
      vertLines: { color: '#f0f0f0' },
      horzLines: { color: '#f0f0f0' }
    },
    crosshair: {
      mode: CrosshairMode.Normal
    },
    rightPriceScale: {
      borderColor: '#f0f0f0'
    },
    timeScale: {
      borderColor: '#f0f0f0',
      timeVisible: true,
      secondsVisible: false
    }
  })

  candleSeries = chart.addCandlestickSeries({
    upColor: '#26a69a',
    downColor: '#ef5350',
    borderVisible: false,
    wickUpColor: '#26a69a',
    wickDownColor: '#ef5350'
  })

  // 加载数据
  candleSeries.setData(props.data)

  // 监听大小变化
  const resizeObserver = new ResizeObserver(entries => {
    for (const entry of entries) {
      chart.applyOptions({
        width: entry.contentRect.width
      })
    }
  })
  resizeObserver.observe(chartContainer.value)
}

watch(() => props.data, (newData) => {
  if (candleSeries) {
    candleSeries.setData(newData)
  }
})

watch(() => props.symbol, () => {
  if (chart) {
    chart.timeScale().fitContent()
  }
})

onMounted(() => {
  initChart()
})

onUnmounted(() => {
  if (chart) {
    chart.remove()
  }
})
</script>

<template>
  <div ref="chartContainer" class="k-line-chart"></div>
</template>

<style scoped lang="scss">
.k-line-chart {
  width: 100%;
  background: #fff;
  border-radius: 8px;
}
</style>
```

## 🔧 组件规范

### 1. 命名规范

- 组件文件：PascalCase，如 `DataTable.vue`
- 组件名称：PascalCase，如 `<DataTable />`
- Props：PascalCase 类型，camelCase 使用

### 2. Props 定义

```typescript
interface Props {
  data: any[]
  loading?: boolean
}

const props = withDefaults(defineProps<Props>(), {
  loading: false
})
```

### 3. Emits 定义

```typescript
interface Emits {
  (e: 'change', value: string): void
  (e: 'submit', data: any): void
}

const emit = defineEmits<Emits>()
```

### 4. Slots 定义

```vue
<template>
  <div class="wrapper">
    <!-- 默认插槽 -->
    <slot />

    <!-- 命名插槽 -->
    <slot name="header" />

    <!-- 作用域插槽 -->
    <slot name="item" :item="item" :index="index" />
  </div>
</template>
```

### 5. 暴露方法

```typescript
// 使用 defineExpose 暴露方法给父组件
defineExpose({
  refresh,
  clear
})
```
