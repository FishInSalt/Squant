# 状态管理设计

## 🎯 概述

本项目使用 **Pinia** 作为状态管理方案，替代 Vuex。

### 为什么选择 Pinia？

- ✅ Vue 3 官方推荐
- ✅ TypeScript 支持完美
- ✅ API 简洁（无需 mutations）
- ✅ 模块化清晰
- ✅ DevTools 支持

## 📦 Store 模块设计

### Store 列表

| Store | 文件路径 | 说明 |
|-------|---------|------|
| `user` | `stores/user.ts` | 用户认证状态、用户信息 |
| `market` | `stores/market.ts` | 行情数据缓存、订阅管理 |
| `strategy` | `stores/strategy.ts` | 策略列表、策略详情 |
| `runtime` | `stores/runtime.ts` | 策略运行状态、执行记录 |
| `websocket` | `stores/websocket.ts` | WebSocket 连接状态 |
| `ui` | `stores/ui.ts` | UI 状态（侧边栏、主题等） |

## 🔐 user.ts - 用户状态

### 状态定义

```typescript
// src/stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { UserInfo, LoginCredentials } from '@/types/user'
import * as authApi from '@/api/auth'

export const useUserStore = defineStore('user', () => {
  // ========== State ==========
  const token = ref<string>('')
  const userInfo = ref<UserInfo | null>(null)
  const permissions = ref<string[]>([])
  const loading = ref(false)

  // ========== Getters ==========
  const isLoggedIn = computed(() => !!token.value)
  const userName = computed(() => userInfo.value?.name || '未登录')
  const hasPermission = computed(() => (permission: string) => {
    return permissions.value.includes(permission)
  })

  // ========== Actions ==========
  /**
   * 登录
   */
  const login = async (credentials: LoginCredentials) => {
    loading.value = true
    try {
      const res = await authApi.login(credentials)
      token.value = res.token
      userInfo.value = res.user
      permissions.value = res.permissions

      // 持久化 token
      localStorage.setItem('token', res.token)
    } finally {
      loading.value = false
    }
  }

  /**
   * 登出
   */
  const logout = async () => {
    try {
      await authApi.logout()
    } finally {
      // 清除状态
      token.value = ''
      userInfo.value = null
      permissions.value = []
      localStorage.removeItem('token')
    }
  }

  /**
   * 获取用户信息
   */
  const fetchUserInfo = async () => {
    const user = await authApi.getUserInfo()
    userInfo.value = user
    permissions.value = user.permissions
  }

  /**
   * 初始化（从 localStorage 恢复）
   */
  const init = () => {
    const savedToken = localStorage.getItem('token')
    if (savedToken) {
      token.value = savedToken
      fetchUserInfo()
    }
  }

  return {
    // State
    token,
    userInfo,
    permissions,
    loading,

    // Getters
    isLoggedIn,
    userName,
    hasPermission,

    // Actions
    login,
    logout,
    fetchUserInfo,
    init
  }
})
```

### 使用示例

```vue
<script setup lang="ts">
import { useUserStore } from '@/stores/user'
import { storeToRefs } from 'pinia'

const userStore = useUserStore()

// 解构 ref 保持响应性
const { isLoggedIn, userName } = storeToRefs(userStore)

const handleLogout = () => {
  userStore.logout()
}
</script>

<template>
  <div v-if="isLoggedIn">
    欢迎, {{ userName }}
    <el-button @click="handleLogout">登出</el-button>
  </div>
</template>
```

## 📊 market.ts - 行情状态

### 状态定义

```typescript
// src/stores/market.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Ticker, KLine, OrderBook } from '@/types/market'
import * as marketApi from '@/api/market'
import { useWebSocketStore } from './websocket'

export const useMarketStore = defineStore('market', () => {
  // ========== State ==========
  const tickers = ref<Map<string, Ticker>>(new Map())
  const kLines = ref<Map<string, KLine[]>>(new Map())
  const orderBooks = ref<Map<string, OrderBook>>(new Map())
  const subscribedSymbols = ref<Set<string>>(new Set())
  const loading = ref(false)

  // ========== Getters ==========
  const getTicker = computed(() => (symbol: string) => {
    return tickers.value.get(symbol)
  })

  const getKLines = computed(() => (symbol: string) => {
    return kLines.value.get(symbol) || []
  })

  const getOrderBook = computed(() => (symbol: string) => {
    return orderBooks.value.get(symbol)
  })

  const allTickers = computed(() => {
    return Array.from(tickers.value.values())
  })

  // ========== Actions ==========
  /**
   * 获取行情列表
   */
  const fetchTickers = async (symbols: string[]) => {
    loading.value = true
    try {
      const data = await marketApi.getTickers(symbols)

      // 更新 Map
      data.forEach(ticker => {
        tickers.value.set(ticker.symbol, ticker)
      })
    } finally {
      loading.value = false
    }
  }

  /**
   * 获取 K 线数据
   */
  const fetchKLines = async (symbol: string, interval: string, limit = 100) => {
    const data = await marketApi.getKLines(symbol, interval, limit)
    kLines.value.set(symbol, data)
  }

  /**
   * 订阅行情
   */
  const subscribeTicker = (symbol: string) => {
    const wsStore = useWebSocketStore()

    if (!subscribedSymbols.value.has(symbol)) {
      wsStore.send({
        type: 'subscribe',
        channel: 'ticker',
        symbol
      })

      subscribedSymbols.value.add(symbol)
    }
  }

  /**
   * 取消订阅
   */
  const unsubscribeTicker = (symbol: string) => {
    const wsStore = useWebSocketStore()

    if (subscribedSymbols.value.has(symbol)) {
      wsStore.send({
        type: 'unsubscribe',
        channel: 'ticker',
        symbol
      })

      subscribedSymbols.value.delete(symbol)
    }
  }

  /**
   * 订阅多个行情
   */
  const subscribeTickers = (symbols: string[]) => {
    symbols.forEach(symbol => subscribeTicker(symbol))
  }

  /**
   * 更新行情数据（从 WebSocket）
   */
  const updateTicker = (data: Ticker) => {
    tickers.value.set(data.symbol, data)
  }

  /**
   * 清除数据
   */
  const clear = () => {
    tickers.value.clear()
    kLines.value.clear()
    orderBooks.value.clear()
    subscribedSymbols.value.clear()
  }

  return {
    // State
    tickers,
    kLines,
    orderBooks,
    subscribedSymbols,
    loading,

    // Getters
    getTicker,
    getKLines,
    getOrderBook,
    allTickers,

    // Actions
    fetchTickers,
    fetchKLines,
    subscribeTicker,
    unsubscribeTicker,
    subscribeTickers,
    updateTicker,
    clear
  }
})
```

## 🎯 strategy.ts - 策略状态

### 状态定义

```typescript
// src/stores/strategy.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Strategy, StrategyConfig } from '@/types/strategy'
import * as strategyApi from '@/api/strategy'

export const useStrategyStore = defineStore('strategy', () => {
  // ========== State ==========
  const strategies = ref<Strategy[]>([])
  const currentStrategy = ref<Strategy | null>(null)
  const loading = ref(false)
  const total = ref(0)

  // ========== Getters ==========
  const activeStrategies = computed(() => {
    return strategies.value.filter(s => s.status === 'active')
  })

  const getStrategyById = computed(() => (id: string) => {
    return strategies.value.find(s => s.id === id)
  })

  const getStrategiesByType = computed(() => (type: string) => {
    return strategies.value.filter(s => s.type === type)
  })

  // ========== Actions ==========
  /**
   * 获取策略列表
   */
  const fetchStrategies = async (params?: {
    page?: number
    pageSize?: number
    type?: string
    status?: string
  }) => {
    loading.value = true
    try {
      const res = await strategyApi.getStrategies(params)
      strategies.value = res.items
      total.value = res.total
    } finally {
      loading.value = false
    }
  }

  /**
   * 获取策略详情
   */
  const fetchStrategy = async (id: string) => {
    const strategy = await strategyApi.getStrategy(id)
    currentStrategy.value = strategy
    return strategy
  }

  /**
   * 创建策略
   */
  const createStrategy = async (config: StrategyConfig) => {
    const strategy = await strategyApi.createStrategy(config)
    strategies.value.unshift(strategy)
    return strategy
  }

  /**
   * 更新策略
   */
  const updateStrategy = async (id: string, config: Partial<StrategyConfig>) => {
    const strategy = await strategyApi.updateStrategy(id, config)
    const index = strategies.value.findIndex(s => s.id === id)
    if (index !== -1) {
      strategies.value[index] = strategy
    }
    if (currentStrategy.value?.id === id) {
      currentStrategy.value = strategy
    }
    return strategy
  }

  /**
   * 删除策略
   */
  const deleteStrategy = async (id: string) => {
    await strategyApi.deleteStrategy(id)
    strategies.value = strategies.value.filter(s => s.id !== id)
    if (currentStrategy.value?.id === id) {
      currentStrategy.value = null
    }
  }

  /**
   * 启用策略
   */
  const enableStrategy = async (id: string) => {
    const strategy = await strategyApi.updateStrategyStatus(id, 'active')
    updateStrategyInList(strategy)
  }

  /**
   * 禁用策略
   */
  const disableStrategy = async (id: string) => {
    const strategy = await strategyApi.updateStrategyStatus(id, 'inactive')
    updateStrategyInList(strategy)
  }

  /**
   * 复制策略
   */
  const duplicateStrategy = async (id: string) => {
    const strategy = await strategyApi.duplicateStrategy(id)
    strategies.value.unshift(strategy)
    return strategy
  }

  /**
   * 更新列表中的策略
   */
  const updateStrategyInList = (strategy: Strategy) => {
    const index = strategies.value.findIndex(s => s.id === strategy.id)
    if (index !== -1) {
      strategies.value[index] = strategy
    }
  }

  return {
    // State
    strategies,
    currentStrategy,
    loading,
    total,

    // Getters
    activeStrategies,
    getStrategyById,
    getStrategiesByType,

    // Actions
    fetchStrategies,
    fetchStrategy,
    createStrategy,
    updateStrategy,
    deleteStrategy,
    enableStrategy,
    disableStrategy,
    duplicateStrategy
  }
})
```

## ⚡ runtime.ts - 运行时状态

### 状态定义

```typescript
// src/stores/runtime.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { RuntimeInstance, ExecutionRecord } from '@/types/runtime'
import * as runtimeApi from '@/api/runtime'

export const useRuntimeStore = defineStore('runtime', () => {
  // ========== State ==========
  const instances = ref<RuntimeInstance[]>([])
  const currentInstance = ref<RuntimeInstance | null>(null)
  const executionRecords = ref<ExecutionRecord[]>([])
  const loading = ref(false)

  // ========== Getters ==========
  const activeInstances = computed(() => {
    return instances.value.filter(i => i.status === 'running')
  })

  const getInstanceByStrategyId = computed(() => (strategyId: string) => {
    return instances.value.find(i => i.strategyId === strategyId)
  })

  const totalPnL = computed(() => {
    return instances.value.reduce((sum, i) => sum + i.pnl, 0)
  })

  // ========== Actions ==========
  /**
   * 获取运行实例列表
   */
  const fetchInstances = async () => {
    loading.value = true
    try {
      instances.value = await runtimeApi.getInstances()
    } finally {
      loading.value = false
    }
  }

  /**
   * 启动策略
   */
  const startStrategy = async (strategyId: string, config: any) => {
    const instance = await runtimeApi.startStrategy(strategyId, config)
    instances.value.unshift(instance)
    return instance
  }

  /**
   * 停止策略
   */
  const stopStrategy = async (instanceId: string) => {
    const instance = await runtimeApi.stopStrategy(instanceId)
    const index = instances.value.findIndex(i => i.id === instanceId)
    if (index !== -1) {
      instances.value[index] = instance
    }
  }

  /**
   * 暂停策略
   */
  const pauseStrategy = async (instanceId: string) => {
    const instance = await runtimeApi.pauseStrategy(instanceId)
    const index = instances.value.findIndex(i => i.id === instanceId)
    if (index !== -1) {
      instances.value[index] = instance
    }
  }

  /**
   * 恢复策略
   */
  const resumeStrategy = async (instanceId: string) => {
    const instance = await runtimeApi.resumeStrategy(instanceId)
    const index = instances.value.findIndex(i => i.id === instanceId)
    if (index !== -1) {
      instances.value[index] = instance
    }
  }

  /**
   * 获取执行记录
   */
  const fetchExecutionRecords = async (instanceId: string) => {
    executionRecords.value = await runtimeApi.getExecutionRecords(instanceId)
  }

  /**
   * 更新实例状态（从 WebSocket）
   */
  const updateInstance = (data: Partial<RuntimeInstance>) => {
    const index = instances.value.findIndex(i => i.id === data.id)
    if (index !== -1) {
      instances.value[index] = { ...instances.value[index], ...data }
    }
  }

  return {
    // State
    instances,
    currentInstance,
    executionRecords,
    loading,

    // Getters
    activeInstances,
    getInstanceByStrategyId,
    totalPnL,

    // Actions
    fetchInstances,
    startStrategy,
    stopStrategy,
    pauseStrategy,
    resumeStrategy,
    fetchExecutionRecords,
    updateInstance
  }
})
```

## 🔌 websocket.ts - WebSocket 状态

### 状态定义

```typescript
// src/stores/websocket.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useWebSocketStore = defineStore('websocket', () => {
  // ========== State ==========
  const ws = ref<WebSocket | null>(null)
  const connected = ref(false)
  const reconnectAttempts = ref(0)
  const maxReconnectAttempts = 5
  const reconnectInterval = 3000 // 3秒

  // ========== Getters ==========
  const isConnecting = computed(() => {
    return ws.value?.readyState === WebSocket.CONNECTING
  })

  // ========== Actions ==========
  /**
   * 连接 WebSocket
   */
  const connect = (url: string) => {
    if (ws.value?.readyState === WebSocket.OPEN) {
      return
    }

    ws.value = new WebSocket(url)

    ws.value.onopen = () => {
      connected.value = true
      reconnectAttempts.value = 0
      console.log('WebSocket connected')
    }

    ws.value.onclose = () => {
      connected.value = false
      console.log('WebSocket disconnected')

      // 自动重连
      if (reconnectAttempts.value < maxReconnectAttempts) {
        reconnectAttempts.value++
        setTimeout(() => connect(url), reconnectInterval)
      }
    }

    ws.value.onerror = (error) => {
      console.error('WebSocket error:', error)
    }

    ws.value.onmessage = (event) => {
      handleMessage(event.data)
    }
  }

  /**
   * 断开连接
   */
  const disconnect = () => {
    ws.value?.close()
    ws.value = null
    connected.value = false
  }

  /**
   * 发送消息
   */
  const send = (message: any) => {
    if (ws.value?.readyState === WebSocket.OPEN) {
      ws.value.send(JSON.stringify(message))
    }
  }

  /**
   * 处理消息
   */
  const handleMessage = (data: string) => {
    const message = JSON.parse(data)

    // 分发到对应的 store
    switch (message.channel) {
      case 'ticker':
        import('./market').then(({ useMarketStore }) => {
          const marketStore = useMarketStore()
          marketStore.updateTicker(message.data)
        })
        break

      case 'runtime':
        import('./runtime').then(({ useRuntimeStore }) => {
          const runtimeStore = useRuntimeStore()
          runtimeStore.updateInstance(message.data)
        })
        break

      default:
        console.warn('Unknown channel:', message.channel)
    }
  }

  return {
    // State
    ws,
    connected,
    reconnectAttempts,

    // Getters
    isConnecting,

    // Actions
    connect,
    disconnect,
    send
  }
})
```

## 🎨 ui.ts - UI 状态

### 状态定义

```typescript
// src/stores/ui.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUiStore = defineStore('ui', () => {
  // ========== State ==========
  const sidebarCollapsed = ref(false)
  const theme = ref<'light' | 'dark'>('light')
  const language = ref<'zh' | 'en'>('zh')
  const loading = ref(false)

  // ========== Getters ==========
  const isDark = computed(() => theme.value === 'dark')

  // ========== Actions ==========
  /**
   * 切换侧边栏
   */
  const toggleSidebar = () => {
    sidebarCollapsed.value = !sidebarCollapsed.value
  }

  /**
   * 设置侧边栏状态
   */
  const setSidebarCollapsed = (collapsed: boolean) => {
    sidebarCollapsed.value = collapsed
  }

  /**
   * 切换主题
   */
  const toggleTheme = () => {
    theme.value = theme.value === 'light' ? 'dark' : 'light'
    applyTheme()
  }

  /**
   * 设置主题
   */
  const setTheme = (newTheme: 'light' | 'dark') => {
    theme.value = newTheme
    applyTheme()
  }

  /**
   * 应用主题
   */
  const applyTheme = () => {
    const html = document.documentElement
    if (theme.value === 'dark') {
      html.classList.add('dark')
    } else {
      html.classList.remove('dark')
    }
  }

  /**
   * 设置语言
   */
  const setLanguage = (lang: 'zh' | 'en') => {
    language.value = lang
  }

  return {
    // State
    sidebarCollapsed,
    theme,
    language,
    loading,

    // Getters
    isDark,

    // Actions
    toggleSidebar,
    setSidebarCollapsed,
    toggleTheme,
    setTheme,
    setLanguage
  }
})
```

## 📝 使用规范

### 1. 在组件中使用 Store

```vue
<script setup lang="ts">
import { useUserStore, useMarketStore } from '@/stores'
import { storeToRefs } from 'pinia'

// 获取 store
const userStore = useUserStore()
const marketStore = useMarketStore()

// 解构 state 和 getters（保持响应性）
const { isLoggedIn, userName } = storeToRefs(userStore)
const { allTickers } = storeToRefs(marketStore)

// 直接解构 actions（不需要响应性）
const { login, logout, fetchTickers } = userStore

// 调用 action
const handleLogin = async () => {
  await login({ username, password })
}
</script>

<template>
  <div>
    <span>{{ userName }}</span>
  </div>
```

## 🔧 最佳实践

### 1. 保持 Store 职责单一

每个 Store 只负责一个领域的状态，避免过于臃肿。

### 2. 使用 TypeScript 类型

确保所有的 state、actions 都有完整的类型定义。

### 3. 合理使用 Getters

Getters 用于派生状态，避免在组件中重复计算。

### 4. 异步 Actions

所有异步操作都放在 Actions 中，确保错误处理。

### 5. 避免 Props drilling

跨组件通信优先使用 Store，而非层层传递 props。
