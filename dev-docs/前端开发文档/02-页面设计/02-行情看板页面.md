# 行情看板页面

## 📋 功能概述

实时展示市场行情数据，包括价格、涨跌幅、成交量、K线图等。

<!-- src/views/market/MarketDashboard.vue -->
<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted, watch } from 'vue'
import { useRouter } from 'vue-router'
import { useMarketStore } from '@/stores/market'
import { useWebSocketStore } from '@/stores/websocket'
import TickerCard from '@/components/market/TickerCard.vue'
import KLineChart from '@/components/charts/KLineChart.vue'
import OrderBook from '@/components/market/OrderBook.vue'
import TradeList from '@/components/market/TradeList.vue'
import { ElSelect, ElOption, ElTag, ElButton, ElTabs, ElTabPane } from 'element-plus'
import type { Ticker } from '@/types/market'

// ========== 状态 ==========
const router = useRouter()
const marketStore = useMarketStore()
const wsStore = useWebSocketStore()

const currentSymbol = ref<string>('BTCUSDT')
const interval = ref<string>('1h')
const loading = ref(false)

// 预设交易对
const symbols = [
  { label: 'BTC/USDT', value: 'BTCUSDT' },
  { label: 'ETH/USDT', value: 'ETHUSDT' },
  { label: 'BNB/USDT', value: 'BNBUSDT' },
  { label: 'SOL/USDT', value: 'SOLUSDT' },
  { label: 'XRP/USDT', value: 'XRPUSDT' }
]

// ========== 计算属性 ==========
const currentTicker = computed(() => {
  return marketStore.getTicker(currentSymbol.value)
})

const kLineData = computed(() => {
  return marketStore.getKLines(currentSymbol.value)
})

// ========== 方法 ==========

/**
 * 加载行情数据
 */
const loadTickers = async () => {
  loading.value = true
  try {
    await marketStore.fetchTickers(symbols.map(s => s.value))
  } finally {
    loading.value = false
  }
}

/**
 * 加载 K 线数据
 */
const loadKLines = async () => {
  await marketStore.fetchKLines(currentSymbol.value, interval.value)
}

/**
 * 切换交易对
 */
const handleSymbolChange = async (symbol: string) => {
  currentSymbol.value = symbol
  await loadKLines()
}

/**
 * 切换时间周期
 */
const handleIntervalChange = async (int: string) => {
  interval.value = int
  await loadKLines()
}

/**
 * 订阅行情
 */
const subscribeMarket = () => {
  const wsUrl = import.meta.env.VITE_WS_BASE_URL
  wsStore.connect(wsUrl)

  // 订阅 ticker
  symbols.forEach(s => {
    marketStore.subscribeTicker(s.value)
  })
}

/**
 * 取消订阅
 */
const unsubscribeMarket = () => {
  symbols.forEach(s => {
    marketStore.unsubscribeTicker(s.value)
  })
  wsStore.disconnect()
}

// ========== 生命周期 ==========
onMounted(() => {
  loadTickers()
  loadKLines()
  subscribeMarket()
})

onUnmounted(() => {
  unsubscribeMarket()
})

// 监听交易对变化
watch(currentSymbol, () => {
  loadKLines()
})
</script>

<template>
  <div class="market-dashboard">
    <!-- 页面头部 -->
    <div class="page-header">
      <div class="header-left">
        <h2 class="page-title">行情看板</h2>
      </div>
      <div class="header-right">
        <ElSelect
          v-model="currentSymbol"
          placeholder="选择交易对"
          @change="handleSymbolChange"
          size="large"
          style="width: 200px"
        >
          <ElOption
            v-for="s in symbols"
            :key="s.value"
            :label="s.label"
            :value="s.value"
          />
        </ElSelect>
      </div>
    </div>

    <!-- 主内容区域 -->
    <div class="dashboard-content">
      <!-- 左侧：行情列表 -->
      <div class="market-list">
        <div class="section-title">
          <h3>自选行情</h3>
        </div>
        <div class="ticker-list">
          <TickerCard
            v-for="s in symbols"
            :key="s.value"
            :ticker="marketStore.getTicker(s.value)"
            :class="{ active: currentSymbol === s.value }"
            @click="handleSymbolChange(s.value)"
          />
        </div>
      </div>

      <!-- 中间：K线图 -->
      <div class="chart-section">
        <!-- 当前交易对信息 -->
        <div class="symbol-info">
          <h3 class="symbol-name">{{ currentSymbol }}</h3>
          <div class="price-info">
            <span class="price">{{ currentTicker?.price || '--' }}</span>

  </div>
</template>

## 🔌 WebSocket 集成

### 订阅 ticker

```typescript
// 在 Market Store 中
const subscribeTicker = (symbol: string) => {
  const wsStore = useWebSocketStore()
  wsStore.send({
    type: 'subscribe',
    channel: 'ticker',
    symbol
  })
}
```

### 接收 ticker 更新

```typescript
// 在 WebSocket Store 中
handleMessage = (data: string) => {
  const message = JSON.parse(data)

  if (message.channel === 'ticker') {
    const marketStore = useMarketStore()
    marketStore.updateTicker(message.data)
  }
}
```
