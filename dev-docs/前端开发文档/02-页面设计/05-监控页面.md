### 1. SystemChart（系统指标图表）

```vue
<!-- components/monitor/SystemChart.vue -->
<script setup lang="ts">
import { computed } from 'vue'

interface Props {
  title: string
  value: number
  color: string
  unit?: string
}

const props = withDefaults(defineProps<Props>(), {
  unit: '%'
})

const strokeDashoffset = computed(() => {
  const radius = 80
  const circumference = 2 * Math.PI * radius
  return circumference - (props.value / 100) * circumference
})
</script>

<template>
  <div class="system-chart">
    <div class="chart-header">
      <h3 class="chart-title">{{ title }}</h3>
      <span class="chart-value">{{ value }}{{ unit }}</span>
    </div>
    <div class="chart-body">
      <svg class="progress-ring" width="200" height="200">
        <circle
          class="progress-ring__circle-bg"
          stroke="#e6e6e6"
          stroke-width="10"
          fill="transparent"
          r="80"
          cx="100"
          cy="100"

          class="progress-ring__circle"
```

### 2. AlertList（告警列表）

```vue
<!-- components/monitor/AlertList.vue -->
<script setup lang="ts">
import { ref, computed } from 'vue'
import { ElTimeline, ElTimelineItem, ElTag, ElButton, ElMessageBox, ElMessage } from 'element-plus'
import { Warning, Info, CircleCheckFilled, CircleCloseFilled } from '@element-plus/icons-vue'

interface Alert {
  id: string
  level: 'error' | 'warning' | 'info'
  message: string
  time: string
  resolved: boolean
}

interface Props {
  alerts: Alert[]
}

interface Emits {
  (e: 'refresh'): void
}

const props = defineProps<Props>()
const emit = defineEmits<Emits>()

const alertIcon = (level: string) => {
  const icons = {
    error: CircleCloseFilled,
    warning: Warning,
    info: Info
  }
  return icons[level as keyof typeof icons] || Info
}

const alertType = (level: string) => {
  const types = {
    error: 'danger',
    warning: 'warning',
    info: 'info'
  }
  return types[level as keyof typeof types] || 'info'
}

const handleResolve = async (id: string) => {
  try {
    await ElMessageBox.confirm('确认标记为已解决？', '提示', {
      type: 'info'
    })
    ElMessage.success('标记成功')
    emit('refresh')
  } catch {
    // 用户取消
  }
}
</script>

<template>
  <div class="alert-list">
    <div class="list-header">
      <h3>告警列表</h3>
      <ElButton size="small" @click="emit('refresh')">刷新</ElButton>
    </div>
    <ElTimeline>
      <ElTimelineItem
        v-for="alert in alerts"
        :key="alert.id"
        :timestamp="alert.time"
        placement="top"
        :icon="alertIcon(alert.level)"
        :type="alertType(alert.level)"
        :hollow="!alert.resolved"
      >
        <div class="alert-item">
          <div class="alert-content">
            <ElTag :type="alertType(alert.level)" size="small">{{ alert.level.toUpperCase() }}</ElTag>
            <span class="alert-message">{{ alert.message }}</span>
          </div>
          <div v-if="!alert.resolved" class="alert-actions">
            <ElButton type="primary" size="small" link @click="handleResolve(alert.id)">
              标记为已解决
            </ElButton>
          </div>
        </div>
      </ElTimelineItem>
    </ElTimeline>
  </div>
</template>

<style scoped lang="scss">
```

### 3. LogViewer（日志查看器）

```vue
<!-- components/monitor/LogViewer.vue -->
<script setup lang="ts">
import { ref, computed } from 'vue'
import { ElButton, ElSelect, ElOption } from 'element-plus'

interface Log {
  timestamp: string
  level: string
  message: string
  module: string
}

interface Props {
  logs: Log[]
}

interface Emits {
  (e: 'refresh'): void
}

const props = defineProps<Props>()
const emit = defineEmits<Emits>()

const filterLevel = ref('all')
const filterModule = ref('all')

const filteredLogs = computed(() => {
  return props.logs.filter(log => {
    if (filterLevel.value !== 'all' && log.level !== filterLevel.value) return false
    if (filterModule.value !== 'all' && log.module !== filterModule.value) return false
    return true
  })
})

const logLevels = ['all', 'DEBUG', 'INFO', 'WARN', 'ERROR']
const modules = ['all', 'API', 'Runtime', 'Market', 'Strategy']

const getLogColor = (level: string) => {
  const colors: Record<string, string> = {
    DEBUG: '#909399',
    INFO: '#409eff',
    WARN: '#e6a23c',
    ERROR: '#f56c6c'
  }
  return colors[level] || '#909399'
}
</script>

<template>
  <div class="log-viewer">
    <div class="toolbar">
      <ElSelect v-model="filterLevel" placeholder="日志级别" size="small">
        <ElOption v-for="level in logLevels" :key="level" :label="level" :value="level" />
      </ElSelect>
      <ElSelect v-model="filterModule" placeholder="模块" size="small">
        <ElOption v-for="mod in modules" :key="mod" :label="mod" :value="mod" />
      </ElSelect>
      <ElButton size="small" @click="emit('refresh')">刷新</ElButton>
    </div>
    <div class="log-container">
      <div v-for="(log, index) in filteredLogs" :key="index" class="log-item">
        <span class="log-timestamp">{{ log.timestamp }}</span>
        <span class="log-level" :style="{ color: getLogColor(log.level) }">[{{ log.level }}]</span>
        <span class="log-module">[{{ log.module }}]</span>
        <span class="log-message">{{ log.message }}</span>
      </div>
    </div>
  </div>
</template>

<style scoped lang="scss">

  }

      }

      }
    }
  }
}
</style>
```
