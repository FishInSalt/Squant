<!-- src/views/runtime/RuntimeDashboard.vue -->
<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted } from 'vue'
import { useRouter } from 'vue-router'
import { useRuntimeStore } from '@/stores/runtime'
import { useWebSocketStore } from '@/stores/websocket'
import { ElTag, ElButton, ElMessage } from 'element-plus'
import RuntimeCard from '@/components/runtime/RuntimeCard.vue'

// ========== 状态 ==========
const router = useRouter()
const runtimeStore = useRuntimeStore()
const wsStore = useWebSocketStore()

const loading = ref(false)

// ========== 计算属性 ==========
const runningInstances = computed(() => runtimeStore.activeInstances)
const pausedInstances = computed(() =>
  runtimeStore.instances.filter(i => i.status === 'paused')
)
const totalPnL = computed(() => runtimeStore.totalPnL.toFixed(2))

// ========== 方法 ==========

/**
 * 加载运行实例
 */
const loadInstances = async () => {
  loading.value = true
  try {
    await runtimeStore.fetchInstances()
  } finally {
    loading.value = false
  }
}

/**
 * 启动策略
 */
const handleStart = async (strategy: any) => {
  try {
    await runtimeStore.startStrategy(strategy.id, strategy.parameters)
    ElMessage.success('策略已启动')
    loadInstances()
  } catch (error: any) {
    ElMessage.error(error.message || '启动失败')
  }
}

/**
 * 暂停策略
 */
const handlePause = async (instanceId: string) => {
  try {
    await runtimeStore.pauseStrategy(instanceId)
    ElMessage.success('策略已暂停')
  } catch (error: any) {
    ElMessage.error(error.message || '暂停失败')
  }
}

/**
 * 恢复策略
 */
const handleResume = async (instanceId: string) => {
  try {
    await runtimeStore.resumeStrategy(instanceId)
    ElMessage.success('策略已恢复')
  } catch (error: any) {
    ElMessage.error(error.message || '恢复失败')
  }
}

/**
 * 停止策略
 */
const handleStop = async (instanceId: string) => {
  try {
    await runtimeStore.stopStrategy(instanceId)
    ElMessage.success('策略已停止')
    loadInstances()
  } catch (error: any) {
    ElMessage.error(error.message || '停止失败')
  }
}

/**
 * 查看详情
 */
const handleViewDetail = (instance: any) => {
  router.push(`/runtime/${instance.id}`)
}

/**
 * 订阅运行状态更新
 */
const subscribeRuntime = () => {
  const wsUrl = import.meta.env.VITE_WS_BASE_URL
  wsStore.connect(wsUrl)
  wsStore.send({
    type: 'subscribe',
    channel: 'runtime'
  })
}

/**
 * 取消订阅
 */
const unsubscribeRuntime = () => {
  wsStore.send({
    type: 'unsubscribe',
    channel: 'runtime'
  })
}

// ========== 生命周期 ==========
onMounted(() => {
  loadInstances()
  subscribeRuntime()
})

onUnmounted(() => {
  unsubscribeRuntime()
})
</script>

<template>
  <div class="runtime-dashboard">
    <!-- 页面头部 -->
    <div class="page-header">
      <h2 class="page-title">策略运行</h2>
    </div>

    <!-- 统计卡片 -->
    <div class="stats-cards">
      <div class="stat-card">
        <div class="stat-label">运行中</div>
        <div class="stat-value">{{ runningInstances.length }}</div>
      </div>
      <div class="stat-card">
        <div class="stat-label">总盈亏</div>
        <div class="stat-value" :class="{ positive: totalPnL > 0, negative: totalPnL < 0 }">
          {{ totalPnL > 0 ? '+' : '' }}${totalPnL}
        </div>
      </div>
      <div class="stat-card">
        <div class="stat-label">平均胜率</div>
        <div class="stat-value">62%</div>
      </div>
      <div class="stat-card">
        <div class="stat-label">总交易次数</div>
        <div class="stat-value">1,234</div>
      </div>
    </div>

    <!-- 运行中实例 -->
    <div class="section">
      <div class="section-header">
        <h3>运行中策略</h3>
        <span class="count">{{ runningInstances.length }}</span>
      </div>
      <div v-if="runningInstances.length > 0" class="instances-grid">
        <RuntimeCard
          v-for="instance in runningInstances"
          :key="instance.id"
          :instance="instance"
          @pause="handlePause"
          @stop="handleStop"
          @detail="handleViewDetail"
        />
      </div>
      <el-empty v-else description="暂无运行中的策略" />
    </div>

    <!-- 暂停实例 -->
    <div class="section">
      <div class="section-header">
        <h3>暂停策略</h3>
        <span class="count">{{ pausedInstances.length }}</span>
      </div>
      <div v-if="pausedInstances.length > 0" class="instances-grid">
        <RuntimeCard
          v-for="instance in pausedInstances"
          :key="instance.id"
          :instance="instance"
          @resume="handleResume"
          @stop="handleStop"
          @detail="handleViewDetail"
        />
      </div>
      <el-empty v-else description="暂无暂停的策略" />
    </div>
  </div>
</template>

<!-- components/runtime/RuntimeCard.vue -->
<script setup lang="ts">
import { computed } from 'vue'
import type { RuntimeInstance } from '@/types/runtime'
import { ElCard, ElTag, ElButton } from 'element-plus'

interface Props {
  instance: RuntimeInstance
}

interface Emits {
  (e: 'pause', instanceId: string): void
  (e: 'resume', instanceId: string): void
  (e: 'stop', instanceId: string): void
  (e: 'detail', instance: RuntimeInstance): void
}

const props = defineProps<Props>()
const emit = defineEmits<Emits>()

const statusType = computed(() => {
  const typeMap: Record<string, any> = {
    running: 'success',
    paused: 'warning',
    stopped: 'info'
  }
  return typeMap[props.instance.status] || 'info'
})

const statusText = computed(() => {
  const textMap: Record<string, string> = {
    running: '运行中',
    paused: '暂停',
    stopped: '已停止'
  }
  return textMap[props.instance.status] || '未知'
})

const pnlClass = computed(() => ({
  positive: props.instance.pnl >= 0,
  negative: props.instance.pnl < 0
}))
</script>

<template>
  <ElCard class="runtime-card" shadow="hover">
    <template #header>
      <div class="card-header">
        <span class="strategy-name">{{ instance.strategyName }}</span>
        <ElTag :type="statusType" size="small">{{ statusText }}</ElTag>
      </div>
      <div class="card-subtitle">{{ instance.symbol }}</div>
    </template>

    <div class="card-body">
      <div class="stat-item">
        <span class="label">盈亏</span>
        <span class="value" :class="pnlClass">
          {{ instance.pnl >= 0 ? '+' : '' }}${instance.pnl.toFixed(2)}
        </span>
      </div>
      <div class="stat-item">
        <span class="label">胜率</span>
        <span class="value">{{ instance.winRate }}%</span>
      </div>
      <div class="stat-item">
        <span class="label">交易次数</span>
        <span class="value">{{ instance.tradeCount }}</span>
      </div>
      <div class="stat-item">
        <span class="label">运行时间</span>
        <span class="value">{{ instance.runTime }}</span>
      </div>
    </div>

    <template #footer>
      <div class="card-actions">
        <ElButton
          v-if="instance.status === 'running'"
          type="warning"
          size="small"
          @click="emit('pause', instance.id)"
        >
          暂停
        </ElButton>
        <ElButton
          v-if="instance.status === 'paused'"
          type="success"
          size="small"
          @click="emit('resume', instance.id)"
        >
          继续
        </ElButton>
        <ElButton
          type="danger"
          size="small"
          @click="emit('stop', instance.id)"
        >
          停止
        </ElButton>
        <ElButton
          type="primary"
          size="small"
          @click="emit('detail', instance)"
        >
          查看详情
        </ElButton>
      </div>
    </template>
  </ElCard>
</template>

<!-- src/views/runtime/RuntimeDetail.vue -->
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRoute } from 'vue-router'
import { useRuntimeStore } from '@/stores/runtime'
import { ElTabs, ElTabPane, ElButton, ElMessage } from 'element-plus'
import PnLChart from '@/components/charts/PnLChart.vue'
import TradeHistory from '@/components/runtime/TradeHistory.vue'

// ========== 状态 ==========
const route = useRoute()
const runtimeStore = useRuntimeStore()

const instanceId = route.params.instanceId as string
const instance = computed(() => runtimeStore.currentInstance)
const activeTab = ref('overview')

// ========== 计算属性 ==========
const isLoading = computed(() => runtimeStore.loading)

// ========== 方法 ==========

/**
 * 加载实例详情
 */
const loadInstance = async () => {
  await runtimeStore.fetchInstance(instanceId)
}

/**
 * 加载执行记录
 */
const loadExecutionRecords = async () => {
  await runtimeStore.fetchExecutionRecords(instanceId)
}

// ========== 生命周期 ==========
onMounted(() => {
  loadInstance()
  loadExecutionRecords()
})
</script>

<template>
  <div class="runtime-detail" v-loading="isLoading">
    <!-- 页面头部 -->
    <div class="page-header">
      <div class="header-left">
        <ElButton @click="() => $router.back()">返回</ElButton>
        <h2 class="page-title">{{ instance?.strategyName }} - 运行详情</h2>
      </div>
      <div class="header-right">
        <ElTag :type="instance?.status === 'running' ? 'success' : 'warning'" size="large">
          {{ instance?.status === 'running' ? '运行中' : '暂停' }}
        </ElTag>
      </div>
    </div>

    <!-- 概览信息 -->
    <div class="overview-section">
      <div class="overview-card">

        <span class="value">{{ instance?.strategyName }}</span>
      </div>

        <span class="value">{{ instance?.symbol }}</span>

        <div class="parameters-view">

        </div>
      </ElTabPane>
    </ElTabs>
  </div>
</template>

## 📊 数据类型

### RuntimeInstance

```typescript
// src/types/runtime.ts
export interface RuntimeInstance {
  id: string
  strategyId: string
  strategyName: string
  symbol: string
  status: 'running' | 'paused' | 'stopped'
  parameters: Record<string, any>
  pnl: number
  winRate: number
  tradeCount: number
  runTime: string
  startTime: string
}

export interface ExecutionRecord {
  id: string
  instanceId: string
  type: 'buy' | 'sell'
  price: number
  amount: number
  timestamp: string
  pnl?: number
}
```
