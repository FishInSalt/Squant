## 💻 代码实现

### 组件代码

```vue
<!-- src/views/strategy/StrategyList.vue -->
<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { useStrategyStore } from '@/stores/strategy'
import { ElButton, ElInput, ElSelect, ElOption, ElTag, ElMessage, ElMessageBox } from 'element-plus'
import DataTable from '@/components/common/DataTable.vue'
import SearchForm from '@/components/common/SearchForm.vue'

// ========== 状态 ==========
const router = useRouter()
const strategyStore = useStrategyStore()

const loading = ref(false)
const selectedIds = ref<string[]>([])

// 搜索表单字段
const searchFields = [
  { prop: 'keyword', label: '关键词', type: 'input', placeholder: '搜索策略名称' },
  { prop: 'status', label: '状态', type: 'select', options: [
    { label: '全部', value: '' },
    { label: '启用', value: 'active' },
    { label: '禁用', value: 'inactive' }
  ]},
  { prop: 'type', label: '类型', type: 'select', options: [
    { label: '全部', value: '' },
    { label: '双均线', value: 'dual_ma' },
    { label: 'MACD', value: 'macd' },
    { label: 'RSI', value: 'rsi' }
  ]}
]

// 搜索参数
const searchParams = ref<Record<string, any>>({})

// ========== 计算属性 ==========
const columns = [
  { prop: 'name', label: '策略名称', width: 200 },
  { prop: 'type', label: '类型', width: 120 },
  { prop: 'symbol', label: '交易对', width: 120 },
  { prop: 'status', label: '状态', width: 100, slot: 'status' },
  { prop: 'createTime', label: '创建时间', width: 180 },
  { prop: 'action', label: '操作', fixed: 'right', width: 250, slot: 'action' }
]

// ========== 方法 ==========

/**
 * 加载策略列表
 */
const loadStrategies = async () => {
  loading.value = true
  try {
    await strategyStore.fetchStrategies(searchParams.value)
  } finally {
    loading.value = false
  }
}

/**
 * 搜索
 */
const handleSearch = (params: any) => {
  searchParams.value = params
  loadStrategies()
}

/**
 * 重置搜索
 */
const handleReset = () => {
  searchParams.value = {}
  loadStrategies()
}

/**
 * 创建策略
 */
const handleCreate = () => {
  router.push('/strategy/create')
}

/**
 * 编辑策略
 */
const handleEdit = (row: any) => {
  router.push(`/strategy/edit/${row.id}`)
}

/**
 * 删除策略
 */
const handleDelete = async (id: string) => {
  try {
    await ElMessageBox.confirm('确定要删除该策略吗？', '提示', {
      type: 'warning'
    })
    await strategyStore.deleteStrategy(id)
    ElMessage.success('删除成功')
    loadStrategies()
  } catch (error) {
    // 用户取消
  }
}

/**
 * 复制策略
 */
const handleDuplicate = async (strategy: any) => {
  try {
    await strategyStore.duplicateStrategy(strategy.id)
    ElMessage.success('复制成功')
    loadStrategies()
  } catch (error: any) {
    ElMessage.error(error.message || '复制失败')
  }
}

/**
 * 启用策略
 */
const handleEnable = async (id: string) => {
  try {
    await strategyStore.enableStrategy(id)
    ElMessage.success('已启用')
  } catch (error: any) {
    ElMessage.error(error.message || '启用失败')
  }
}

/**
 * 禁用策略
 */
const handleDisable = async (id: string) => {
  try {
    await strategyStore.disableStrategy(id)
    ElMessage.success('已禁用')
  } catch (error: any) {
    ElMessage.error(error.message || '禁用失败')
  }
}

/**
 * 批量删除
 */
const handleBatchDelete = async () => {
  if (selectedIds.value.length === 0) {
    ElMessage.warning('请先选择要删除的策略')
    return
  }

  try {
    await ElMessageBox.confirm(`确定要删除选中的 ${selectedIds.value.length} 个策略吗？`, '提示', {
      type: 'warning'
    })

    await Promise.all(selectedIds.value.map(id => strategyStore.deleteStrategy(id)))
    ElMessage.success('批量删除成功')
    loadStrategies()
  } catch (error) {
    // 用户取消
  }
}

// ========== 生命周期 ==========
onMounted(() => {
  loadStrategies()
})
</script>

<template>
  <div class="strategy-list">
    <!-- 页面头部 -->
    <div class="page-header">
      <h2 class="page-title">策略管理</h2>
      <div class="header-actions">
        <ElButton type="danger" :disabled="selectedIds.length === 0" @click="handleBatchDelete">
          批量删除
        </ElButton>
        <ElButton type="primary" @click="handleCreate">
          + 新建策略
        </ElButton>
      </div>
    </div>

    <!-- 搜索表单 -->
    <SearchForm
      :fields="searchFields"
      @search="handleSearch"
      @reset="handleReset"
    />

    <!-- 策略列表 -->
    <DataTable
      :data="strategyStore.strategies"
      :columns="columns"
      :loading="loading"
      :pagination="{
        page: 1,
        pageSize: 20,
        total: strategyStore.total
      }"
      :selectable="true"
      @selection-change="(selection) => selectedIds = selection.map((item: any) => item.id)"
      @page-change="loadStrategies"
    >
      <template #status="{ row }">
        <ElTag :type="row.status === 'active' ? 'success' : 'info'" size="small">
          {{ row.status === 'active' ? '启用' : '禁用' }}
        </ElTag>
      </template>
      <template #action="{ row }">
        <ElButton
          v-if="row.status === 'inactive'"
          type="success"
          link
          size="small"
          @click="handleEnable(row.id)"
        >
          启用
        </ElButton>
        <ElButton
          v-if="row.status === 'active'"
          type="warning"
          link
          size="small"
          @click="handleDisable(row.id)"
        >
          禁用
        </ElButton>
        <ElButton type="primary" link size="small" @click="handleEdit(row)">
          编辑
        </ElButton>
        <ElButton type="info" link size="small" @click="handleDuplicate(row)">
          复制
        </ElButton>
        <ElButton type="danger" link size="small" @click="handleDelete(row.id)">
          删除
        </ElButton>
      </template>
    </DataTable>
  </div>
</template>

<style scoped lang="scss">
.strategy-list {
  .page-header {
    display: flex;
    justify-content: space-between;
    align-items: center;
    margin-bottom: 20px;

    .page-title {
      margin: 0;
      font-size: 24px;
      font-weight: 600;
    }

    .header-actions {
      display: flex;
      gap: 12px;
    }
  }
}
</style>
```

## 📝 策略表单页面

### 组件代码

```vue
<!-- src/views/strategy/StrategyForm.vue -->
<script setup lang="ts">
import { ref, reactive, computed, onMounted } from 'vue'
import { useRouter, useRoute } from 'vue-router'
import { useStrategyStore } from '@/stores/strategy'
import { ElForm, ElFormItem, ElInput, ElSelect, ElOption, ElButton, ElMessage } from 'element-plus'
import ParameterEditor from '@/components/strategy/ParameterEditor.vue'
import type { FormInstance, FormRules } from 'element-plus'

// ========== 状态 ==========
const router = useRouter()
const route = useRoute()
const strategyStore = useStrategyStore()

const formRef = ref<FormInstance>()
const loading = ref(false)
const isEdit = computed(() => !!route.params.id)

// 表单数据
const formData = reactive({
  name: '',
  type: 'dual_ma',
  symbol: 'BTCUSDT',
  parameters: {
    fastPeriod: 5,
    slowPeriod: 20,
    lotSize: 0.01
  },
  description: ''
})

// 表单验证规则
const rules: FormRules = {
  name: [
    { required: true, message: '请输入策略名称', trigger: 'blur' },
    { min: 2, max: 50, message: '策略名称长度在 2 到 50 个字符', trigger: 'blur' }
  ],
  type: [
    { required: true, message: '请选择策略类型', trigger: 'change' }
  ],
  symbol: [
    { required: true, message: '请选择交易对', trigger: 'change' }
  ]
}

// 策略类型选项
const strategyTypes = [
  { label: '双均线策略', value: 'dual_ma' },
  { label: 'MACD策略', value: 'macd' },
  { label: 'RSI策略', value: 'rsi' }
]

// 交易对选项
const symbols = [
  { label: 'BTC/USDT', value: 'BTCUSDT' },
  { label: 'ETH/USDT', value: 'ETHUSDT' },
  { label: 'BNB/USDT', value: 'BNBUSDT' }
]

// ========== 方法 ==========

/**
 * 加载策略详情
 */
const loadStrategy = async () => {
  const id = route.params.id as string
  const strategy = await strategyStore.fetchStrategy(id)

  formData.name = strategy.name
  formData.type = strategy.type
  formData.symbol = strategy.symbol
  formData.parameters = strategy.parameters
  formData.description = strategy.description
}

/**
 * 保存
 */
const handleSave = async () => {
  if (!formRef.value) return

  const valid = await formRef.value.validate().catch(() => false)
  if (!valid) return

  loading.value = true
  try {
    if (isEdit.value) {
      await strategyStore.updateStrategy(route.params.id as string, formData)
      ElMessage.success('更新成功')
    } else {
      await strategyStore.createStrategy(formData)
      ElMessage.success('创建成功')
    }
    router.push('/strategy/list')
  } catch (error: any) {
    ElMessage.error(error.message || '保存失败')
  } finally {
    loading.value = false
  }
}

/**
 * 取消
 */
const handleCancel = () => {
  router.back()
}

// ========== 生命周期 ==========
onMounted(() => {
  if (isEdit.value) {
    loadStrategy()
  }
})
</script>

<template>
  <div class="strategy-form">
    <!-- 页面头部 -->
    <div class="page-header">
      <h2 class="page-title">{{ isEdit ? '编辑策略' : '创建策略' }}</h2>
    </div>

    <!-- 表单 -->
    <ElForm
      ref="formRef"
      :model="formData"
      :rules="rules"
      label-width="120px"
    >
      <ElFormItem label="策略名称" prop="name">
        <ElInput v-model="formData.name" placeholder="请输入策略名称" />
      </ElFormItem>

      <ElFormItem label="策略类型" prop="type">
        <ElSelect v-model="formData.type" placeholder="请选择策略类型">
          <ElOption
            v-for="type in strategyTypes"
            :key="type.value"
            :label="type.label"
            :value="type.value"
          />
        </ElSelect>
      </ElFormItem>

      <ElFormItem label="交易对" prop="symbol">
        <ElSelect v-model="formData.symbol" placeholder="请选择交易对">
          <ElOption
            v-for="sym in symbols"
            :key="sym.value"
            :label="sym.label"
            :value="sym.value"
          />
        </ElSelect>
      </ElFormItem>

      <ElFormItem label="策略参数">
        <ParameterEditor
          v-model="formData.parameters"
          :type="formData.type"
        />
      </ElFormItem>

      <ElFormItem label="描述">
        <ElInput
          v-model="formData.description"
          type="textarea"
          :rows="4"
          placeholder="请输入策略描述"
        />
      </ElFormItem>

      <ElFormItem>
        <ElButton :loading="loading" type="primary" @click="handleSave">
          保存
        </ElButton>
        <ElButton @click="handleCancel">
          取消
        </ElButton>
      </ElFormItem>
    </ElForm>
  </div>
</template>

<style scoped lang="scss">
.strategy-form {
  max-width: 800px;

  .page-header {
    margin-bottom: 24px;

    .page-title {
      margin: 0;
      font-size: 24px;
      font-weight: 600;
    }
  }
}
</style>
```

## 🧩 参数编辑器组件

```vue
<!-- components/strategy/ParameterEditor.vue -->
<script setup lang="ts">
import { computed, watch } from 'vue'
import { ElForm, ElFormItem, ElInput, ElInputNumber } from 'element-plus'

interface Props {
  modelValue: Record<string, any>
  type: string
}

const props = defineProps<Props>()
const emit = defineEmits<{
  (e: 'update:modelValue', value: Record<string, any>): void
}>()

// 参数配置
const parameterConfigs: Record<string, any[]> = {
  dual_ma: [
    { key: 'fastPeriod', label: '快速均线周期', type: 'number', min: 1, max: 100 },
    { key: 'slowPeriod', label: '慢速均线周期', type: 'number', min: 1, max: 200 },
    { key: 'lotSize', label: '下单数量', type: 'number', min: 0.001, step: 0.001 }
  ],
  macd: [
    { key: 'fastPeriod', label: '快线周期', type: 'number', min: 1, max: 100 },
    { key: 'slowPeriod', label: '慢线周期', type: 'number', min: 1, max: 200 },
    { key: 'signalPeriod', label: '信号线周期', type: 'number', min: 1, max: 100 },
    { key: 'lotSize', label: '下单数量', type: 'number', min: 0.001, step: 0.001 }
  ],
  rsi: [
    { key: 'period', label: 'RSI周期', type: 'number', min: 1, max: 100 },
    { key: 'overbought', label: '超买阈值', type: 'number', min: 50, max: 100 },
    { key: 'oversold', label: '超卖阈值', type: 'number', min: 0, max: 50 },
    { key: 'lotSize', label: '下单数量', type: 'number', min: 0.001, step: 0.001 }
  ]
}

const currentConfigs = computed(() => {
  return parameterConfigs[props.type] || []
})

const updateParameter = (key: string, value: any) => {
  emit('update:modelValue', {
    ...props.modelValue,
    [key]: value
  })
}
</script>

<template>
  <div class="parameter-editor">
    <ElForm label-width="140px">
      <ElFormItem
        v-for="config in currentConfigs"
        :key="config.key"
        :label="config.label"
      >
        <ElInputNumber
          v-if="config.type === 'number'"
          :model-value="modelValue[config.key]"
          :min="config.min"
          :max="config.max"
          :step="config.step || 1"
          @update:model-value="(val) => updateParameter(config.key, val)"
        />
      </ElFormItem>
    </ElForm>
  </div>
</template>

<style scoped lang="scss">
.parameter-editor {
  :deep(.el-form-item) {
    margin-bottom: 12px;
  }
}
</style>
```
