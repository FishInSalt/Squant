## 4. 策略接口设计

### 4.1 策略模板 (Python)

```python
"""
策略模板文件
用户基于此模板实现自定义策略
"""

from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import List, Dict, Optional
from datetime import datetime
from decimal import Decimal
import logging

# 策略SDK (由系统提供)
from squant.strategy_sdk import (
    Context,
    Bar,
    Tick,
    Order,
    OrderType,
    OrderSide,
    OrderStatus,
    Position,
)

logger = logging.getLogger(__name__)


@dataclass
class StrategyConfig:
    """策略配置参数"""
    # 在策略启动时传入
    symbols: List[str]  # 交易对列表,如 ["BTCUSDT", "ETHUSDT"]
    account_id: str  # 交易所账户ID
    run_mode: str  # 'backtest', 'paper', 'live'

    # 策略自定义参数
    param1: float = 0.0
    param2: int = 0
    # ... 更多参数


class BaseStrategy(ABC):
    """
    策略基类
    所有自定义策略必须继承此类并实现抽象方法
    """

    def __init__(self, config: StrategyConfig, context: Context):
        """
        策略初始化

        Args:
            config: 策略配置参数
            context: 策略上下文 (提供行情、交易、日志等API)
        """
        self.config = config
        self.ctx = context
        logger.info(f"策略初始化: {self.__class__.__name__}, 配置: {config}")

    @abstractmethod
    async def on_init(self):
        """
        策略初始化回调
        在策略启动时调用一次,用于初始化策略状态
        """
        pass

    @abstractmethod
    async def on_bar(self, bar: Bar):
        """
        K线数据回调

        Args:
            bar: K线数据
        """
        pass

    async def on_tick(self, tick: Tick):
        """
        逐笔行情回调 (可选)

        Args:
            tick: 逐笔行情数据
        """
        pass

    async def on_order(self, order: Order):
        """
        订单状态更新回调 (可选)

        Args:
            order: 订单对象
        """
        pass

    async def on_position(self, position: Position):
        """
        持仓变化回调 (可选)

        Args:
            position: 持仓对象
        """
        pass

    async def on_error(self, error: Exception):
        """
        错误回调 (可选)

        Args:
            error: 异常对象
        """
        logger.error(f"策略错误: {error}")

    async def on_stop(self):
        """
        策略停止回调
        在策略停止时调用一次,用于清理资源
        """
        logger.info("策略停止")


# ============ 示例策略 ============

class SimpleMovingAverageStrategy(BaseStrategy):
    """
    简单均线策略示例
    当短期均线上穿长期均线时买入,下穿时卖出
    """

    async def on_init(self):
        self.short_period = 5
        self.long_period = 20
        self.price_history: Dict[str, List[float]] = {}

        for symbol in self.config.symbols:
            self.price_history[symbol] = []

        logger.info(f"均线策略初始化完成, 短期={self.short_period}, 长期={self.long_period}")

    async def on_bar(self, bar: Bar):
        symbol = bar.symbol

        # 记录价格
        self.price_history[symbol].append(float(bar.close))

        # 保持历史数据长度
        if len(self.price_history[symbol]) > self.long_period:
            self.price_history[symbol].pop(0)

        # 计算均线
        if len(self.price_history[symbol]) >= self.long_period:
            short_ma = sum(self.price_history[symbol][-self.short_period:]) / self.short_period
            long_ma = sum(self.price_history[symbol][-self.long_period:]) / self.long_period

            # 获取当前持仓
            positions = await self.ctx.get_positions(symbol)
            current_position = positions[0] if positions else None

            # 交易逻辑
            if short_ma > long_ma and (not current_position or current_position.quantity <= 0):
                # 金叉买入
                await self.ctx.place_order(
                    symbol=symbol,
                    side=OrderSide.BUY,
                    order_type=OrderType.MARKET,
                    quantity=Decimal("0.001")  # 买入0.001 BTC
                )
                logger.info(f"买入信号: {symbol}, 短期MA={short_ma:.2f}, 长期MA={long_ma:.2f}")

            elif short_ma < long_ma and current_position and current_position.quantity > 0:
                # 死叉卖出
                await self.ctx.place_order(
                    symbol=symbol,
                    side=OrderSide.SELL,
                    order_type=OrderType.MARKET,
                    quantity=current_position.quantity
                )
                logger.info(f"卖出信号: {symbol}, 短期MA={short_ma:.2f}, 长期MA={long_ma:.2f}")

    async def on_stop(self):
        logger.info("均线策略停止, 清理资源")


# 策略入口函数 (系统调用)
def create_strategy(config: StrategyConfig, context: Context) -> BaseStrategy:
    """
    策略工厂函数

    Args:
        config: 策略配置
        context: 策略上下文

    Returns:
        策略实例
    """
    return SimpleMovingAverageStrategy(config, context)
```

### 4.2 策略SDK (Context)

⚠️ **重要更新**: Context类已简化为接口分离设计,降低策略开发者学习成本。

**设计理念:**
- 使用接口隔离原则 (ISP),将功能划分为三个独立API
- 降低策略开发者学习成本,避免过度封装
- 提供更清晰的职责划分

```python
"""
策略SDK - 提供给策略使用的API
"""

from decimal import Decimal
from datetime import datetime
from squant.models import Bar, Tick, Order, OrderType, OrderSide, Position, Balance
from .api import MarketAPI, TradingAPI, LogAPI


class Context:
    """
    策略上下文
    提供行情、交易、日志等API给策略使用

    接口分离设计:
    - ctx.market: 行情API
    - ctx.trading: 交易API
    - ctx.log: 日志API
    """

    def __init__(
        self,
        strategy_id: str,
        execution_id: str,
        account_id: str,
        run_mode: str,
        market_api: 'MarketAPI',
        trading_api: 'TradingAPI',
        log_api: 'LogAPI'
    ):
        self.strategy_id = strategy_id
        self.execution_id = execution_id
        self.account_id = account_id
        self.run_mode = run_mode

        # 接口分离: 策略直接使用 self.ctx.market/ trading/ log
        self.market = market_api
        self.trading = trading_api
        self.log = log_api

    # ============ 辅助属性 ============

    @property
    def run_mode(self) -> str:
        """获取运行模式"""
        return self._run_mode

    @run_mode.setter
    def run_mode(self, value: str):
        self._run_mode = value

    @property
    def account_id(self) -> str:
        """获取账户ID"""
        return self._account_id

    @account_id.setter
    def account_id(self, value: str):
        self._account_id = value

    @property
    def current_time(self) -> datetime:
        """获取当前时间"""
        return datetime.now()


# ============ 接口定义 ============

class MarketAPI(ABC):
    """行情API接口"""

    @abstractmethod
    async def get_current_bar(self, symbol: str, interval: str = '1m') -> Optional[Bar]:
        """获取当前K线"""
        pass

    @abstractmethod
    async def get_historical_bars(
        self,
        symbol: str,
        interval: str = '1m',
        limit: int = 100
    ) -> List[Bar]:
        """获取历史K线"""
        pass

    @abstractmethod
    async def subscribe_bar(
        self,
        symbol: str,
        interval: str = '1m',
        callback: Callable[[Bar], None]
    ):
        """订阅K线数据"""
        pass

    @abstractmethod
    async def subscribe_tick(
        self,
        symbol: str,
        callback: Callable[[Tick], None]
    ):
        """订阅逐笔行情 (可选)"""
        pass


class TradingAPI(ABC):
    """交易API接口"""

    @abstractmethod
    async def place_order(
        self,
        symbol: str,
        side: OrderSide,
        order_type: OrderType,
        quantity: Decimal,
        price: Optional[Decimal] = None,
        stop_price: Optional[Decimal] = None,
        client_order_id: Optional[str] = None
    ) -> Order:
        """下单"""
        pass

    @abstractmethod
    async def cancel_order(self, order_id: str) -> bool:
        """撤单"""
        pass

    @abstractmethod
    async def get_order(self, order_id: str) -> Optional[Order]:
        """查询订单"""
        pass

    @abstractmethod
    async def get_positions(self, symbol: Optional[str] = None) -> List[Position]:
        """查询持仓"""
        pass

    @abstractmethod
    async def get_balance(self) -> List[Balance]:
        """查询账户余额"""
        pass


class LogAPI(ABC):
    """日志API接口"""

    @abstractmethod
    def info(self, message: str, **kwargs):
        """记录info级别日志"""
        pass

    @abstractmethod
    def debug(self, message: str, **kwargs):
        """记录debug级别日志"""
        pass

    @abstractmethod
    def warning(self, message: str, **kwargs):
        """记录warning级别日志"""
        pass

    @abstractmethod
    def error(self, message: str, **kwargs):
        """记录error级别日志"""
        pass
```

**策略使用示例:**

```python
# 旧设计 (过度封装)
await self.ctx.get_current_bar("BTCUSDT")
await self.ctx.place_order(...)
self.ctx.info("买入信号")

# 新设计 (接口分离)
await self.ctx.market.get_current_bar("BTCUSDT")
await self.ctx.trading.place_order(...)
self.ctx.log.info("买入信号")
```

**优势:**
1. ✅ 清晰的职责划分 (market/ trading/ log)
2. ✅ 降低学习成本 (开发者只需关注对应的API)
3. ✅ 符合接口隔离原则 (ISP)
4. ✅ 更好的代码提示和类型安全
