## 3. 数据模型设计

### 3.1 PostgreSQL 数据库设计

#### 3.1.1 核心表结构

```sql
-- 交易所账户表
CREATE TABLE accounts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL,
    exchange VARCHAR(50) NOT NULL,  -- 'binance', 'okx', etc.
    account_name VARCHAR(100),
    api_key_encrypted BYTEA NOT NULL,  -- AES-256加密
    api_secret_encrypted BYTEA NOT NULL,  -- AES-256加密
    passphrase_encrypted BYTEA,  -- 某些交易所需要
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_accounts_user_id ON accounts(user_id);
CREATE INDEX idx_accounts_exchange ON accounts(exchange);

-- 自选币种表
CREATE TABLE watchlist (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL,
    symbol VARCHAR(20) NOT NULL,  -- 'BTCUSDT', 'ETHUSDT', etc.
    exchange VARCHAR(50) NOT NULL,
    added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id, symbol, exchange)
);

CREATE INDEX idx_watchlist_user_id ON watchlist(user_id);

-- 策略表
CREATE TABLE strategies (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    language VARCHAR(20) DEFAULT 'python',  -- 'python', 'javascript'
    version VARCHAR(20) DEFAULT '1.0.0',
    file_path VARCHAR(500),  -- 策略文件路径 (MinIO对象存储)
    metadata JSONB,  -- 策略元数据 (作者、参数说明等)
    is_validated BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_strategies_user_id ON strategies(user_id);
CREATE INDEX idx_strategies_name ON strategies(name);

-- 策略版本表
CREATE TABLE strategy_versions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    strategy_id UUID NOT NULL REFERENCES strategies(id) ON DELETE CASCADE,
    version VARCHAR(20) NOT NULL,
    file_path VARCHAR(500),
    changelog TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(strategy_id, version)
);

-- 策略执行记录表
CREATE TABLE strategy_executions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    strategy_id UUID NOT NULL REFERENCES strategies(id),
    account_id UUID REFERENCES accounts(id),
    run_mode VARCHAR(20) NOT NULL,  -- 'backtest', 'paper', 'live'
    status VARCHAR(20) NOT NULL,  -- 'running', 'stopped', 'error'
    config JSONB,  -- 运行配置 (参数、交易对等)
    container_id VARCHAR(100),  -- Docker容器ID
    start_time TIMESTAMP,
    end_time TIMESTAMP,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_executions_strategy_id ON strategy_executions(strategy_id);
CREATE INDEX idx_executions_status ON strategy_executions(status);
CREATE INDEX idx_executions_run_mode ON strategy_executions(run_mode);

 -- 订单表
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),  -- 内部主键（UUID，无业务含义）
    exchange_order_id VARCHAR(50) NOT NULL,  -- 交易所订单ID（业务ID）
    execution_id UUID REFERENCES strategy_executions(id),
    account_id UUID NOT NULL REFERENCES accounts(id),
    exchange VARCHAR(50) NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    side VARCHAR(10) NOT NULL,  -- 'buy', 'sell'
    order_type VARCHAR(20) NOT NULL,  -- 'market', 'limit', 'stop_limit'
    price DECIMAL(20, 8),
    quantity DECIMAL(20, 8),
    filled_quantity DECIMAL(20, 8) DEFAULT 0,
    status VARCHAR(20) NOT NULL,  -- 'new', 'partial_filled', 'filled', 'canceled', 'rejected'
    is_simulation BOOLEAN DEFAULT FALSE,  -- 是否模拟订单
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    -- 复合唯一约束：确保同一交易所的订单ID唯一
    CONSTRAINT uq_orders_exchange_order_id UNIQUE (exchange, exchange_order_id)
);

CREATE INDEX idx_orders_execution_id ON orders(execution_id);
CREATE INDEX idx_orders_account_id ON orders(account_id);
CREATE INDEX idx_orders_symbol ON orders(symbol);
CREATE INDEX idx_orders_status ON orders(status);
CREATE INDEX idx_orders_created_at ON orders(created_at);
CREATE INDEX idx_orders_exchange_order_id ON orders(exchange_order_id);  -- 便于通过交易所订单ID查询

-- 持仓表
CREATE TABLE positions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    account_id UUID NOT NULL REFERENCES accounts(id),
    execution_id UUID REFERENCES strategy_executions(id),
    symbol VARCHAR(20) NOT NULL,
    side VARCHAR(10) NOT NULL,  -- 'long', 'short'
    quantity DECIMAL(20, 8) NOT NULL,
    entry_price DECIMAL(20, 8),
    current_price DECIMAL(20, 8),
    unrealized_pnl DECIMAL(20, 8) DEFAULT 0,
    is_simulation BOOLEAN DEFAULT FALSE,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(account_id, execution_id, symbol, is_simulation)
);

CREATE INDEX idx_positions_account_id ON positions(account_id);
CREATE INDEX idx_positions_execution_id ON positions(execution_id);

-- 策略日志表
CREATE TABLE strategy_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    execution_id UUID NOT NULL REFERENCES strategy_executions(id),
    level VARCHAR(10) NOT NULL,  -- 'debug', 'info', 'warning', 'error'
    message TEXT NOT NULL,
    context JSONB,  -- 附加上下文信息
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_logs_execution_id ON strategy_logs(execution_id);
CREATE INDEX idx_logs_level ON strategy_logs(level);
CREATE INDEX idx_logs_created_at ON strategy_logs(created_at);

-- 回测结果表
CREATE TABLE backtest_results (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    execution_id UUID NOT NULL REFERENCES strategy_executions(id),
    strategy_id UUID NOT NULL REFERENCES strategies(id),
    total_return DECIMAL(10, 4),
    annual_return DECIMAL(10, 4),
    max_drawdown DECIMAL(10, 4),
    sharpe_ratio DECIMAL(10, 4),
    win_rate DECIMAL(5, 4),
    total_trades INT,
    profit_trades INT,
    loss_trades INT,
    start_date DATE,
    end_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_backtest_strategy_id ON backtest_results(strategy_id);

-- 告警表
CREATE TABLE alerts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    execution_id UUID REFERENCES strategy_executions(id),
    type VARCHAR(50) NOT NULL,  -- 'error', 'warning', 'info'
    severity VARCHAR(20) NOT NULL,  -- 'low', 'medium', 'high', 'critical'
    title VARCHAR(200) NOT NULL,
    message TEXT NOT NULL,
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_alerts_execution_id ON alerts(execution_id);
CREATE INDEX idx_alerts_is_read ON alerts(is_read);
CREATE INDEX idx_alerts_created_at ON alerts(created_at);

-- 行情历史数据表 (K线)
CREATE TABLE klines (
    id BIGSERIAL PRIMARY KEY,
    exchange VARCHAR(50) NOT NULL,
    symbol VARCHAR(20) NOT NULL,
    interval VARCHAR(10) NOT NULL,  -- '1m', '5m', '15m', '1h', '4h', '1d', etc.
    open_time TIMESTAMP NOT NULL,
    close_time TIMESTAMP NOT NULL,
    open_price DECIMAL(20, 8) NOT NULL,
    high_price DECIMAL(20, 8) NOT NULL,
    low_price DECIMAL(20, 8) NOT NULL,
    close_price DECIMAL(20, 8) NOT NULL,
    volume DECIMAL(20, 8) NOT NULL,
    quote_volume DECIMAL(20, 8),
    trades_count INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(exchange, symbol, interval, open_time)
);

-- 使用 TimescaleDB 创建超表以优化时序数据查询
-- SELECT create_hypertable('klines', 'open_time', chunk_time_interval => INTERVAL '1 day');

CREATE INDEX idx_klines_symbol_interval ON klines(symbol, interval);
CREATE INDEX idx_klines_open_time ON klines(open_time DESC);
```

### 3.2 Redis 数据结构设计

```python
# 1. 实时行情缓存 (Hash, TTL: 5s)
Key: market:ticker:{exchange}:{symbol}
Value: {
    "price": "45000.50",
    "change_24h": "+2.5%",
    "volume_24h": "1234567890",
    "timestamp": "1706073600"
}

# 2. 热门币种列表 (List)
Key: market:trending
Value: ["BTCUSDT", "ETHUSDT", "BNBUSDT", ...]

# 3. 用户会话 (Hash)
Key: session:{session_id}
Value: {
    "user_id": "uuid",
    "expires_at": "1706080000"
}

# 4. 消息队列流 (Redis Streams)
Key: stream:market_data
Key: stream:order_events
Key: stream:strategy_logs

# 5. 策略运行状态 (Hash)
Key: runtime:execution:{execution_id}
Value: {
    "status": "running",
    "container_id": "abc123",
    "start_time": "1706073600",
    "cpu_usage": "12.5",
    "memory_usage": "256"
}

# 6. 分布式锁 (String)
Key: lock:strategy:{execution_id}
Value: "locked"
TTL: 30s
```
