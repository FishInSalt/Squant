## 5. 安全性设计

### 5.1 API Key 安全存储

```python
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.backends import default_backend
import os
import base64


class APIKeyEncryptor:
    """API Key加密/解密器"""

    def __init__(self, master_key_env: str = "MASTER_KEY"):
        """
        初始化加密器

        Args:
            master_key_env: 存储主密钥的环境变量名
        """
        self.master_key = os.getenv(master_key_env)
        if not self.master_key:
            raise ValueError(f"环境变量 {master_key_env} 未设置")

        # 从Base64解码主密钥
        self.master_key = base64.b64decode(self.master_key.encode())

    def encrypt(self, plaintext: str) -> str:
        """
        加密明文

        Args:
            plaintext: 明文 (如API Key)

        Returns:
            Base64编码的密文
        """
        # AES-256-GCM加密
        aesgcm = AESGCM(self.master_key)
        nonce = os.urandom(12)  # 96位nonce
        ciphertext = aesgcm.encrypt(nonce, plaintext.encode(), None)

        # 拼接nonce和ciphertext,并Base64编码
        combined = nonce + ciphertext
        return base64.b64encode(combined).decode()

    def decrypt(self, ciphertext: str) -> str:
        """
        解密密文

        Args:
            ciphertext: Base64编码的密文

        Returns:
            明文
        """
        # Base64解码
        combined = base64.b64decode(ciphertext.encode())

        # 分离nonce和ciphertext
        nonce = combined[:12]
        ciphertext_bytes = combined[12:]

        # AES-256-GCM解密
        aesgcm = AESGCM(self.master_key)
        plaintext = aesgcm.decrypt(nonce, ciphertext_bytes, None)

        return plaintext.decode()


# 使用示例
encryptor = APIKeyEncryptor()

# 加密存储
api_key = "your_binance_api_key"
encrypted = encryptor.encrypt(api_key)
# 保存到数据库: accounts.api_key_encrypted = encrypted

# 使用时解密
decrypted = encryptor.decrypt(encrypted)
# 使用 decrypted 调用交易所API
```

**⚠️ 环境变量方式的安全风险:**
1. 进程列表可能泄露 (`ps aux` | grep MASTER_KEY)
2. 日志可能意外记录环境变量
3. 密钥轮换困难
4. 多环境管理不便

### 5.1.2 推荐方案: 使用keyring密钥环 (适用于个人项目)

### 5.2 策略代码安全检查

```python
import ast
from typing import List, Set


class SecurityValidator:
    """策略代码安全验证器"""

    # 禁止导入的模块
    FORBIDDEN_MODULES = {
        'os',
        'subprocess',
        'shutil',
        'sys',
        'socket',
        'http',
        'urllib',
        'requests',  # 策略不应直接发HTTP请求
    }

    # 允许的模块白名单
    ALLOWED_MODULES = {
        'math',
        'statistics',
        'datetime',
        'decimal',
        'typing',
        'dataclasses',
        'pandas',  # 数据分析
        'numpy',   # 数值计算
        'squant',  # 策略SDK
    }

    def validate(self, code: str) -> tuple[bool, List[str]]:
        """
        验证策略代码安全性

        Args:
            code: 策略代码

        Returns:
            (是否通过, 错误消息列表)
        """
        errors = []

        try:
            tree = ast.parse(code)
        except SyntaxError as e:
            return False, [f"语法错误: {e}"]

        # 检查导入语句
        for node in ast.walk(tree):
            if isinstance(node, ast.Import):
                for alias in node.names:
                    module_name = alias.name.split('.')[0]
                    if module_name in self.FORBIDDEN_MODULES:
                        errors.append(f"禁止导入模块: {module_name}")
                    elif module_name not in self.ALLOWED_MODULES:
                        errors.append(f"未授权模块: {module_name}")

            elif isinstance(node, ast.ImportFrom):
                module_name = node.module.split('.')[0] if node.module else ''
                if module_name in self.FORBIDDEN_MODULES:
                    errors.append(f"禁止从模块导入: {module_name}")
                elif module_name not in self.ALLOWED_MODULES:
                    errors.append(f"未授权模块导入: {module_name}")

            # 检查函数调用 (防止直接调用危险函数)
            elif isinstance(node, ast.Call):
                if isinstance(node.func, ast.Name):
                    func_name = node.func.id
                    if func_name in ['exec', 'eval', 'open']:
                        errors.append(f"禁止调用函数: {func_name}")

        return len(errors) == 0, errors
```

**⚠️ 静态检查的局限性:**
- 无法发现动态导入 (`__import__(user_input)`)
- 无法检测代码注入 (`eval(f"{user_input}")`)
- 无法防止无限循环
- 无法检测递归深度攻击

### 5.2.2 分层安全策略 (运行时安全)

为弥补静态检查的不足,实现**多层安全策略**:

```python
import asyncio
import resource
from typing import List


class StrategyRuntimeSecurity:
    """策略运行时安全管理器"""

    def __init__(self, max_cpu_percent: float = 50.0, max_memory_mb: int = 512):
        """
        初始化安全管理器

        Args:
            max_cpu_percent: 最大CPU使用率(%)
            max_memory_mb: 最大内存使用量(MB)
        """
        self.max_cpu_percent = max_cpu_percent
        self.max_memory_mb = max_memory_mb
        self.timeout_seconds = 300  # 默认超时5分钟

    async def run_with_timeout(
        self,
        strategy_func,
        timeout_seconds: int = None
    ):
        """
        策略运行超时控制

        Args:
            strategy_func: 策略函数
            timeout_seconds: 超时时间(秒)

        Returns:
            策略执行结果

        Raises:
            asyncio.TimeoutError: 超时异常
        """
        timeout = timeout_seconds or self.timeout_seconds

        try:
            return await asyncio.wait_for(strategy_func(), timeout=timeout)
        except asyncio.TimeoutError:
            logger.error(f"策略运行超时 ({timeout}秒), 强制终止")
            raise

    def set_resource_limits(self):
        """
        设置资源限制 (防止资源耗尽)

        仅适用于Linux/MacOS系统
        """
        import sys
        import resource

        try:
            # 设置最大CPU时间 (单位: 秒)
            # 防止无限循环
            resource.setrlimit(
                resource.RLIMIT_CPU,
                (self.timeout_seconds, self.timeout_seconds * 1.1)
            )

            # 设置最大内存 (单位: 字节)
            max_memory_bytes = self.max_memory_mb * 1024 * 1024
            resource.setrlimit(
                resource.RLIMIT_AS,
                (max_memory_bytes, max_memory_bytes * 1.1)
            )

            logger.info(f"资源限制设置: CPU={self.timeout_seconds}s, Memory={self.max_memory_mb}MB")

        except (OSError, ValueError) as e:
            logger.warning(f"设置资源限制失败: {e}")

    def check_resource_usage(self) -> dict:
        """
        检查当前资源使用情况

        Returns:
            资源使用统计
        """
        import psutil

        process = psutil.Process()

        cpu_percent = process.cpu_percent(interval=1)
        memory_info = process.memory_info()

        return {
            "cpu_percent": cpu_percent,
            "memory_mb": memory_info.rss / 1024 / 1024,
            "memory_percent": (memory_info.rss / psutil.virtual_memory().total) * 100,
            "num_threads": process.num_threads(),
            "num_fds": process.num_fds() if hasattr(process, 'num_fds') else 0
        }

    def is_resource_exceeded(self) -> tuple[bool, str]:
        """
        检查资源是否超限

        Returns:
            (是否超限, 超限原因)
        """
        usage = self.check_resource_usage()

        if usage["cpu_percent"] > self.max_cpu_percent * 2:  # 允许2倍短期峰值
            return True, f"CPU使用率过高: {usage['cpu_percent']:.1f}%"

        if usage["memory_mb"] > self.max_memory_mb:
            return True, f"内存使用超限: {usage['memory_mb']:.1f}MB"

        if usage["num_threads"] > 100:
            return True, f"线程数过多: {usage['num_threads']}"

        return False, ""
```

**完整的安全策略总结:**

| 层级 | 安全措施 | 检测/防护对象 |
|------|---------|--------------|
| **静态检查** | AST语法分析 | 危险模块、危险函数 |
| **运行时沙箱** | Docker容器隔离 | 文件系统、网络、进程 |
| **资源限制** | CPU/Memory配额 | 资源耗尽、无限循环 |
| **超时控制** | asyncio.wait_for | 无限循环、长时间运行 |
| **网络隔离** | --network=none | 未授权网络访问 |
| **只读挂载** | volumes:ro | 策略代码篡改 |
| **非root运行** | user:1000:1000 | 容器提权 |

### 5.3 认证与授权

```python
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from jose import JWTError, jwt
from datetime import datetime, timedelta


# JWT配置
SECRET_KEY = "your-secret-key"  # 从环境变量读取
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 60 * 24 * 7  # 7天

security = HTTPBearer()


def create_access_token(data: dict) -> str:
    """创建JWT访问令牌"""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    to_encode.update({"exp": expire})
    encoded_jwt = jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)
    return encoded_jwt


def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)) -> str:
    """验证JWT令牌并返回用户ID"""
    try:
        payload = jwt.decode(
            credentials.credentials,
            SECRET_KEY,
            algorithms=[ALGORITHM]
        )
        user_id: str = payload.get("sub")
        if user_id is None:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="无效的认证凭据"
            )
        return user_id
    except JWTError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="无效的认证凭据"
        )


# 依赖注入示例
async def get_current_user(user_id: str = Depends(verify_token)) -> User:

    if not user:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail="用户不存在"
        )
    return user
```
