        """获取行情缓存"""
        key = f"market:ticker:{exchange}:{symbol}"
        data = await self.redis.get(key)
        if data:
            return json.loads(data)
        return None

    async def set_ticker(self, exchange: str, symbol: str, data: dict, ttl: int = 5):
        """设置行情缓存"""
        key = f"market:ticker:{exchange}:{symbol}"
        await self.redis.setex(key, ttl, json.dumps(data))

    async def invalidate_watchlist(self, user_id: str):
        """失效用户自选缓存"""
        pattern = f"watchlist:{user_id}:*"
        keys = await self.redis.keys(pattern)
        if keys:
            await self.redis.delete(*keys)

## 9. 扩展性设计

### 9.1 市场扩展 (港股/美股)

```python
# 交易所适配器抽象 (已在2.3.3中定义)
class ExchangeAdapter(ABC):
    """交易所适配器抽象基类"""
    # ...

# 具体实现
class HKEXAdapter(ExchangeAdapter):
    """港交所适配器"""
    # ...

class NYSEAdapter(ExchangeAdapter):
    """纽交所适配器"""
    # ...

# 工厂模式
class ExchangeAdapterFactory:
    """交易所适配器工厂"""

    adapters = {
        'binance': BinanceAdapter,
        'okx': OKXAdapter,
        'hkex': HKEXAdapter,      # 港股
        'nyse': NYSEAdapter,      # 美股
    }

    @classmethod
    def create(cls, exchange: str, config: dict) -> ExchangeAdapter:
        """创建交易所适配器"""
        adapter_class = cls.adapters.get(exchange)
        if not adapter_class:
            raise ValueError(f"不支持的交易所: {exchange}")
        return adapter_class(config)
```
